class Compute 
{
    static Pair getMinMax(long a[], long n)  
    {
       
        //Write your code here
         long min=100000000,max=0;
    for (int i=0;i<n;i++){
        if(max<a[i]){
            max=a[i];
        }if(min>a[i]){
            min=a[i];
        }
    }
    
   
    Pair p1 = new Pair(min,max );
    return p1;
    }
}
